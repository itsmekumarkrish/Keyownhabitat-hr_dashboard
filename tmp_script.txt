        function startSeamlessInterview(roleName) {
            currentRole = roleName;
            document.getElementById('current-role-badge').textContent = roleName;
            document.getElementById('form-role-title').textContent = roleName;
            
            // Populate Job Description
            document.getElementById('jd-content').innerHTML = roleJDs[roleName] || "Details available upon request.";
            document.getElementById('jd-accordion').removeAttribute('open');
            
            // Reset form
            document.getElementById('job-application-form').reset();
            
            // Push history state so browser Back Button works perfectly
            if (window.location.hash !== '#apply') {
                history.pushState({ view: 'form' }, "", window.location.href.split('#')[0] + "#apply");
            }
            
            // Force scroll to top immediately before transition
            window.scrollTo(0, 0);
            
            // Transition Views
            document.getElementById('job-selection-view').classList.add('fade-out');
            
            setTimeout(() => {
                document.getElementById('seamless-chat-view').style.display = 'block';
                document.getElementById('seamless-chat-view').classList.add('active');
                
                // Now cleanly scroll to the top of the view
                window.scrollTo({ top: 0, behavior: 'smooth' });
                
            }, 500); // Wait for fade-out
        }

        function submitApplication(event) {
            event.preventDefault();
            
            const btn = event.target.querySelector('button[type="submit"]');
            const originalText = btn.innerHTML;
            btn.innerHTML = '<i class="ph ph-spinner ph-spin"></i> Submitting...';
            btn.disabled = true;
            
            // Simulate API call
            setTimeout(() => {
                alert(`Thank you for applying to the ${currentRole} position! We have received your application successfully.`);
                backToJobs();
                btn.innerHTML = originalText;
                btn.disabled = false;
            }, 1500);
        }

        window.addEventListener('popstate', (event) => {
            // When browser back is clicked, hash changes away from #apply
            if (window.location.hash !== '#apply') {
                const seamlessView = document.getElementById('seamless-chat-view');
                if (seamlessView && seamlessView.classList.contains('active')) {
                    executeBackToJobsVisuals();
                }
            }
        });

        function backToJobs(event) {
            if (event) event.preventDefault();
            
            // If the URL has #apply, clear it safely without navigating away
            if (window.location.hash === '#apply') {
                history.pushState("", document.title, window.location.pathname + window.location.search);
            }
            
            executeBackToJobsVisuals();
        }
        
        function executeBackToJobsVisuals() {
            document.getElementById('seamless-chat-view').style.opacity = '0';
            
            setTimeout(() => {
                document.getElementById('seamless-chat-view').classList.remove('active');
                document.getElementById('seamless-chat-view').style.display = 'none';
                document.getElementById('seamless-chat-view').style.opacity = '';
                
                const jobView = document.getElementById('job-selection-view');
                jobView.classList.remove('fade-out');
                // Force visibility in case CSS class fails
                jobView.style.display = 'block';
                jobView.style.opacity = '1';
                jobView.style.pointerEvents = 'auto';
                jobView.style.position = 'relative';
                
                // Ensure page is scrollable and return to top reliably
                document.body.style.overflow = 'auto';
                document.documentElement.style.overflow = 'auto';
                document.getElementById('jd-accordion').removeAttribute('open');
                
                // Use a harsh scroll to prevent interruption
                window.scrollTo(0, 0);
                setTimeout(() => {
                    window.scrollTo(0, 0);
                    // Also aggressively scroll the view into place
                    document.getElementById('job-selection-view').scrollIntoView({ behavior: 'auto', block: 'start' });
                }, 50);
            }, 400);
        }
        
        window.addEventListener('scroll', () => {
            const navbar = document.querySelector('.navbar');
            if (window.scrollY > 50) {
                navbar.classList.add('scrolled');
            } else {
                navbar.classList.remove('scrolled');
            }
        });
    </script>
